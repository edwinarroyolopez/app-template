const STARTER_ROUTES = [
  {
    href: '/dashboard',
    label: 'Resumen',
    icon: LayoutDashboard,
    section: 'account' as const,
    highlight: true,
  },
  {
    href: '/workspace',
    label: 'Datos y formularios',
    icon: PenLine,
    section: 'navigation' as const,
  },
  {
    href: '/patterns',
    label: 'Accesos y estados',
    icon: SlidersHorizontal,
    section: 'navigation' as const,
  },
] as const;

export default function ProtectedLayout({ children }: { children: ReactNode }) {
  const router = useRouter();
  const pathname = usePathname();
  const activeWorkspace = useWorkspaceStore((s) => s.activeWorkspace);
  const { token, user, capabilities, logout, isHydrated } = useAuthStore();
  useMe();

  const [isSidebarOpen, setIsSidebarOpen] = useState(false);

  useEffect(() => {
    if (!isHydrated) return;
    if (!token) {
      router.replace('/login');
    }
  }, [isHydrated, token, router]);

  const handleLogout = () => {
    logout();
    useWorkspaceStore.getState().setActiveWorkspace(null);
    router.replace('/login');
  };

  const isPremiumTier = capabilities ? !capabilities.hasTierLimits : true;

  const routes = STARTER_ROUTES.map((route) => ({
    href: route.href,
    label: route.label,
    icon: route.icon,
    section: route.section,
    highlight: 'highlight' in route ? Boolean(route.highlight) : false,
    locked: route.href === '/workspace' && !isPremiumTier,
    badge: route.href === '/workspace' && !isPremiumTier ? 'PLUS' : undefined,
  }));

  if (!isHydrated) return null;
  if (!token) return null;

  const sideNavActiveContext =
    activeWorkspace != null
      ? {
          href: '/dashboard',
          title: 'Workspace activo',
          name: activeWorkspace.name,
          icon: LayoutDashboard,
        }
      : null;

  return (
    <Shell>
      <SideNav
        open={isSidebarOpen}
        onClose={() => setIsSidebarOpen(false)}
        pathname={pathname}
        routes={routes}
        user={{ name: user?.name, role: user?.role }}
        onSignOut={handleLogout}
        activeContext={sideNavActiveContext}
        footerNote="template-web starter"
      />

      <ContentWrapper>
        <TopBar>
          <div style={{ display: 'flex', alignItems: 'center', gap: '12px', minWidth: 0 }}>
            <MenuButton onClick={() => setIsSidebarOpen(true)} type="button" aria-label="Abrir menú">
              <MenuIcon size={20} />
            </MenuButton>
            <Brand>Starter shell</Brand>
          </div>

          <div style={{ display: 'flex', alignItems: 'center', gap: '12px', minWidth: 0 }}>
            <WorkspaceSwitcher />
            <div
              style={{
                fontSize: '11px',
                color: theme.colors.muted,
                fontWeight: 600,
                textTransform: 'uppercase',
                letterSpacing: '0.06em',
              }}
            >
              {user?.role ?? '—'}
            </div>
          </div>
        </TopBar>

        <Main>{children}</Main>
      </ContentWrapper>

      {isSidebarOpen && (
        <Backdrop type="button" onClick={() => setIsSidebarOpen(false)} aria-label="Cerrar menú" />
      )}
    </Shell>
  );
}

async function fetchDemoRows(): Promise<{ id: string; title: string }[]> {
  await new Promise((r) => setTimeout(r, 450));
  return [
    { id: 'a', title: 'Registro de ejemplo A' },
    { id: 'b', title: 'Registro de ejemplo B' },
  ];
}

export default function WorkspaceDemoPage() {
  const [title, setTitle] = useState('');
  const [saved, setSaved] = useState<string | null>(null);
  const capabilities = useAuthStore((s) => s.capabilities);
  const locked = capabilities?.hasTierLimits === true;

  const { data, isFetching, refetch, isError } = useQuery({
    queryKey: ['starter-workspace-demo'],
    queryFn: fetchDemoRows,
  });

  if (locked) {
    return (
      <Page>
        <Card>
          <div style={{ display: 'flex', gap: 12, alignItems: 'flex-start' }}>
            <Lock size={22} color={theme.colors.muted} style={{ flexShrink: 0, marginTop: 2 }} />
            <div>
              <Title style={{ fontSize: '1.1rem', marginBottom: 8 }}>Ruta con tier limitado</Title>
              <Muted>
                En este starter, <code>/workspace</code> aparece bloqueada en la navegación cuando{' '}
                <code>capabilities.hasTierLimits</code> es verdadero (mapeado desde la API). Sustituye la regla por la de tu
                producto.
              </Muted>
            </div>
          </div>
        </Card>
      </Page>
    );
  }

  return (
    <Page>
      <div>
        <Title>Datos y formularios</Title>
        <Muted style={{ marginTop: 10 }}>
          Estado local en React, envío simulado y lista cargada con React Query (sin API de negocio). Úsalo como plantilla para
          pantallas CRUD.
        </Muted>
      </div>

      <RefreshHeader
        title="Lista demo"
        subtitle="React Query · refetch manual"
        isFetching={isFetching}
        onRefresh={() => refetch()}
      />

      <Card>
        <form
          onSubmit={(e) => {
            e.preventDefault();
            setSaved(title.trim() || '(vacío)');
            setTitle('');
          }}
          style={{ display: 'grid', gap: 14 }}
        >
          <label style={{ fontSize: 12, fontWeight: 600, color: theme.colors.muted }}>Título</label>
          <Input value={title} onChange={(e) => setTitle(e.target.value)} placeholder="Escribe y guarda (local)" />
          <Button type="submit" disabled={isFetching}>
            Guardar en estado local
          </Button>
          {saved !== null ? (
            <Muted style={{ fontSize: 13 }}>
              Último valor guardado: <strong style={{ color: theme.colors.text }}>{saved}</strong>
            </Muted>
          ) : null}

export interface ImageViewerModalProps {
    visible: boolean;
    imageUrl?: string;
    onClose: () => void;
    alt?: string;
}

export function ImageViewerModal({ visible, imageUrl, onClose, alt = 'Imagen' }: ImageViewerModalProps) {
    useEffect(() => {
        if (!visible) return;

        const onKeyDown = (event: KeyboardEvent) => {
            if (event.key === 'Escape') {
                onClose();
            }
        };

        window.addEventListener('keydown', onKeyDown);
        return () => window.removeEventListener('keydown', onKeyDown);
    }, [visible, onClose]);

    if (!visible || !imageUrl) return null;

    return (
        <Overlay onClick={onClose}>
            <Content onClick={(event) => event.stopPropagation()}>
                <CloseButton type="button" onClick={onClose} aria-label="Cerrar imagen">
                    <X size={18} />
                </CloseButton>
                <Image src={imageUrl} alt={alt} />
            </Content>
        </Overlay>
    );
}

export type ModalProps = {
  open: boolean;
  title?: string;
  onClose: () => void;
  children?: React.ReactNode;
  /** Fila de acciones predefinida (p. ej. ModalActions) */
  actions?: React.ReactNode;
  /** Pie personalizado (formularios); se muestra junto a `actions` si ambos existen */
  footer?: React.ReactNode;
  /** narrow ≈ 400px (antes AppModal); default ≈ 520px */
  size?: "default" | "narrow";
  /** soft = overlay más claro + blur (perfil anterior AppModal) */
  backdrop?: "default" | "soft";
};

export const Modal: React.FC<ModalProps> = ({
  open,
  title,
  onClose,
  children,
  actions,
  footer,
  size = "default",
  backdrop = "default",
}) => {
  useEffect(() => {
    if (!open) return;
    const onKey = (e: KeyboardEvent) => e.key === "Escape" && onClose();
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  }, [open, onClose]);

  if (!open) return null;

  const bar = actions || footer ? (
    <Footer>
      {actions}
      {footer}
    </Footer>
  ) : null;

  return (
    <Overlay $soft={backdrop === "soft"} onClick={onClose}>
      <Dialog
        role="dialog"
        aria-modal="true"
        aria-labelledby={title ? "modal-title" : undefined}
        $narrow={size === "narrow"}
        onClick={(e) => e.stopPropagation()}
      >
        <Header>
          <Title id="modal-title">{title}</Title>
          <CloseButton type="button" aria-label="Cerrar" onClick={onClose}>
            ×
          </CloseButton>
        </Header>
        <Body>{children}</Body>
        {bar}
      </Dialog>
    </Overlay>
  );
};

"use client";
import React from "react";
import { Grid, Cell } from "./NumberSelector.styles";

export interface NumberSelectorProps {
  value?: number;
  onChange: (n: number) => void;
  max?: number;
  pad?: number;
  disabled?: boolean;
  disabledNumbers?: Set<number> | number[];
}

export const NumberSelector: React.FC<NumberSelectorProps> = ({
  value,
  onChange,
  max = 999,
  pad = 3,
  disabled = false,
  disabledNumbers,
}) => {
  const disabledSet = React.useMemo<Set<number>>(() => {
    if (!disabledNumbers) return new Set<number>();
    return disabledNumbers instanceof Set
      ? disabledNumbers
      : new Set(disabledNumbers);
  }, [disabledNumbers]);

  const items = React.useMemo(
    () => Array.from({ length: max + 1 }, (_, i) => i),
    [max]
  );
  const isDisabled = (n: number) => disabled || disabledSet.has(n);

  const refGrid = React.useRef<HTMLDivElement>(null);
  const colGuess = 10;

  const findNext = React.useCallback(
    (from: number, delta: number) => {
      if (from == null) return undefined;
      let i = from;
      for (let step = 0; step <= max; step++) {
        i = i + delta;
        if (i < 0 || i > max) break;
        if (!isDisabled(i)) return i;
      }
      return undefined;
    },
    [max, isDisabled]
  );

  const onKeyDown = (e: React.KeyboardEvent<HTMLDivElement>) => {
    if (disabled) return;
    if (value == null) return;

    let next: number | undefined;
    switch (e.key) {
      case "ArrowRight":
        next = findNext(value, +1);
        break;
      case "ArrowLeft":
        next = findNext(value, -1);
        break;
      case "ArrowDown":
        next = findNext(value, +colGuess);
        break;
      case "ArrowUp":
        next = findNext(value, -colGuess);
        break;
      case "Home":
        next = items.find((n) => !isDisabled(n));
        break;
      case "End":
        for (let i = max; i >= 0; i--) {
          if (!isDisabled(i)) {
            next = i;
            break;
          }
        }
        break;
      case "Enter":
      case " ":
        break;
      default:
        return;
    }
    if (typeof next === "number") {
      e.preventDefault();
      onChange(next);
    }
  };

  return (
    <Grid
      tabIndex={0}
      aria-disabled={disabled}
      role="grid"
      onKeyDown={onKeyDown}
      ref={refGrid}
      aria-label="Selector de números"
    >
      {items.map((n) => {
        const label = n.toString().padStart(pad, "0");
        const active = value === n;
        const blocked = isDisabled(n);

        return (
          <Cell
            key={n}
            type="button"
            $active={active}
            $disabled={blocked}
            onClick={() => !blocked && onChange(n)}
            aria-pressed={active}
            aria-disabled={blocked}
            disabled={blocked}
            title={
              blocked ? `Número ${label} no disponible` : `Elegir ${label}`
            }
          >
            {label}
          </Cell>
        );
      })}
    </Grid>
  );
};

`;

function normalizeOptions<T extends string>(
    options: ReadonlyArray<T | TimeRangeOption<T>>,
): ReadonlyArray<TimeRangeOption<T>> {
    return options.map((opt) => (typeof opt === "string" ? { label: opt, value: opt } : opt));
}

export function TimeRangeSelector<T extends string>({
    value,
    onChange,
    options = DEFAULT_TIME_RANGES as unknown as ReadonlyArray<T | TimeRangeOption<T>>,
    ariaLabel = "Time range selector",
    disabled,
    size = "md",
    className,
}: Props<T>) {
    const normalized = React.useMemo(() => normalizeOptions(options), [options]);

    function move(step: number) {
        const idx = normalized.findIndex((o) => o.value === value);
        if (idx < 0) return;

        const nextIdx = Math.min(Math.max(idx + step, 0), normalized.length - 1);
        const next = normalized[nextIdx]?.value;
        if (next && next !== value) onChange(next);
    }

    return (
        <Tabs
            className={className}
            role="tablist"
            aria-label={ariaLabel}
            aria-disabled={disabled ? "true" : "false"}
            $disabled={disabled}
            $size={size}
            onKeyDown={(e) => {
                if (disabled) return;
                if (e.key === "ArrowLeft") move(-1);
                if (e.key === "ArrowRight") move(1);
            }}
        >
            {normalized.map((opt) => {
                const active = opt.value === value;
                return (
                    <TabBtn
                        key={opt.value}
                        type="button"
                        role="tab"
                        aria-selected={active}
                        $active={active}
                        $size={size}
                        onClick={() => !disabled && onChange(opt.value)}
                    >
                        {opt.label}
                    </TabBtn>
                );
            })}
        </Tabs>
    );
}

// src/lib/fetcher.ts
"use client";

import { useAuthStore } from "@/modules/auth/store";

export const API_BASE =
    process.env.NEXT_PUBLIC_API_BASE || "http://localhost:7000/api";

/**
 * Hace un fetch puro pero con headers básicos.
 */
async function coreFetch<T>(
    path: string,
    init: RequestInit = {},
    token?: string
): Promise<T> {
    const headers = new Headers(init.headers);
    headers.set("Content-Type", "application/json");

    // token explícito > token del store
    const storeToken = useAuthStore.getState().accessToken;
    const finalToken = token || storeToken;

    if (finalToken) {
        headers.set("Authorization", `Bearer ${finalToken}`);
    }

    const res = await fetch(`${API_BASE}${path}`, {
        ...init,
        headers,
        cache: "no-store",
    });

    // respuesta OK
    if (res.ok) {
        try {
            return (await res.json()) as T;
        } catch {
            return undefined as T;
        }
    }

    // manejo 401 → limpiar sesión
    if (res.status === 401) {
        useAuthStore.getState().clear?.();
        throw new Error("Sesión expirada. Inicia sesión nuevamente.");
    }

    // cualquier otro error → mensaje genérico
    let message = `Error ${res.status}`;
    try {
        const json = await res.json();
        if (json?.message) message = json.message;
    } catch { }

    throw new Error(message);
}

/**
 * Wrapper principal que replica la idea del ZIP
 */
export async function http<T = unknown>(
    path: string,
    init: RequestInit = {},
    token?: string
): Promise<T> {
    return coreFetch<T>(path, init, token);
}

const REASON_MAP: Record<string, StarterCapabilityReason> = {
  PREMIUM_REQUIRED: 'TIER_REQUIRED',
  NOT_OWNER: 'NOT_OWNER',
  ACCOUNT_INACTIVE: 'ACCOUNT_INACTIVE',
  MAX_MINES_REACHED: 'LIMIT_WORKSPACES',
  MAX_MEMBERS_REACHED: 'LIMIT_MEMBERS',
  MAX_LAVADAS_REACHED: 'LIMIT_PRIMARY_RECORDS',

  FEATURE_DISABLED: 'FEATURE_DISABLED',
};

function mapReason(r?: string): StarterCapabilityReason | undefined {
  if (!r) return undefined;
  return REASON_MAP[r] ?? 'TIER_REQUIRED';
}

function mapCapabilities(raw: {
  canInviteMembers: { enabled: boolean; reason?: string };
  canCreateWorkspace: { enabled: boolean; reason?: string };
  canUseApi: { enabled: boolean; reason?: string };
  hasTierLimits: boolean;
}): AccountCapabilities {
  return {
    canCreateMultipleWorkspaces: {
      enabled: raw.canCreateWorkspace.enabled,
      reason: mapReason(raw.canCreateWorkspace.reason),
    },
    canInviteMembers: {
      enabled: raw.canInviteMembers.enabled,
      reason: mapReason(raw.canInviteMembers.reason),
    },
    canUseApi: {
      enabled: raw.canUseApi.enabled,
      reason: mapReason(raw.canUseApi.reason),
    },
    hasTierLimits: raw.hasTierLimits,
  };
}

export function useMe() {
  const hydrateSession = useAuthStore((s) => s.hydrateSession);
  const token = useAuthStore((s) => s.token);

  const query = useQuery({
    queryKey: ['me'],
    queryFn: getMe,
    enabled: !!token,
  });

  useEffect(() => {
    if (!query.data) return;

    const caps = query.data.accountCapabilities ?? query.data.capabilities;
    if (!caps) return;

    hydrateSession({
      user: {
        id: query.data.user.id,
        phone: query.data.user.phone,
        name: query.data.user.name ?? '',
        role: query.data.user.role as any,
        employeeRole: query.data.user.employeeRole as any,
      },
      workspaces: query.data.workspaces ?? [],
      capabilities: mapCapabilities(caps),
      accountPermissions: (query.data.account.permissions || []).filter(
        Boolean,
      ) as AccountPermissionKey[],
      usage: {
        workspaces: query.data.usage.workspaces,
        members: query.data.usage.members,
      },
    });
  }, [query.data, hydrateSession]);

  return query;
}

import { persist, createJSONStorage } from 'zustand/middleware';

// User del backend: usualmente un líder
export type User = {
    _id: string;
    name: string;
    phone: string;
    role?: string; // 'leader' | 'admin' | 'super_admin'
} | null;

interface AuthState {
    accessToken: string | null;
    refreshToken: string | null;
    user: User;
    setTokens: (a: string, r?: string) => void;
    setUser: (u: User) => void;
    clear: () => void;
}

export const useAuthStore = create<AuthState>()(
    persist(
        (set) => ({
            accessToken: null,
            refreshToken: null,
            user: null,
            setTokens: (a, r) =>
                set({
                    accessToken: a,
                    refreshToken: r ?? null,
                }),
            setUser: (u) => set({ user: u }),
            clear: () => set({ accessToken: null, refreshToken: null, user: null }),
        }),
        {
            name: 'credigas-auth', // clave en localStorage
            storage: createJSONStorage(() => localStorage),
        }
    )
);

// src/stores/auth.store.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';

export type UserRole =
  | 'OWNER'
  | 'OWNER_VIEWER'
  | 'ADMIN'
  | 'MEMBER'
  | 'VIEWER';

export type AuthUser = {
  id: string;
  phone: string;
  name: string;
  role: UserRole;
  employeeRole?: 'BUSINESS_MANAGER' | 'SELLER' | 'MANUFACTURER' | 'TRANSPORTER' | 'GENERAL_SUPPORT';
};

export type WorkspaceSummary = {
  id: string;
  name: string;
};

export type StarterCapabilityReason =
  | 'TIER_REQUIRED'
  | 'NOT_OWNER'
  | 'ACCOUNT_INACTIVE'
  | 'LIMIT_WORKSPACES'
  | 'LIMIT_MEMBERS'
  | 'LIMIT_PRIMARY_RECORDS'
  | 'TIER_LIMIT_REACHED'
  | 'FEATURE_DISABLED';

export type CapabilityFlag = {
  enabled: boolean;
  reason?: StarterCapabilityReason;
};

/** Mapped from `template-backend` `/auth/me` `accountCapabilities`. */
export type AccountCapabilities = {
  canCreateMultipleWorkspaces: CapabilityFlag;
  canInviteMembers: CapabilityFlag;
  canUseApi: CapabilityFlag;
  hasTierLimits: boolean;
};

export type AccountPermissionKey =
  | 'PLATFORM_ADMIN'
  | 'BILLING_MANAGE'
  | 'MEMBER_INVITES'
  | 'DATA_EXPORT';

export type AccountUsage = {
  workspaces: { used: number; max: number };
  members: { used: number; max: number };
};

type AuthState = {
  token: string | null;
  user: AuthUser | null;
  workspaces: WorkspaceSummary[];
  capabilities: AccountCapabilities | null;
  accountPermissions: AccountPermissionKey[];
  usage: AccountUsage | null;
  isAuthenticated: boolean;
  isHydrated: boolean;

  setSession: (data: {
    token: string;
    user: AuthUser;
    workspaces?: WorkspaceSummary[];
    capabilities?: AccountCapabilities | null;
    accountPermissions?: AccountPermissionKey[];
    usage?: AccountUsage | null;
  }) => void;

  hydrateSession: (data: {
    user: AuthUser;
    workspaces?: WorkspaceSummary[];
    capabilities?: AccountCapabilities | null;
    accountPermissions?: AccountPermissionKey[];
    usage?: AccountUsage | null;
  }) => void;

  logout: () => void;
};

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      token: null,
      user: null,
      workspaces: [],
      capabilities: null,
      accountPermissions: [],
      usage: null,
      isAuthenticated: false,
      isHydrated: false,

      setSession: ({
        token,
        user,
        workspaces = [],
        capabilities = null,
        accountPermissions,
        usage = null,
      }) =>
        set((state) => ({
          token,
          user,
          workspaces,
          capabilities: capabilities ?? state.capabilities,
          accountPermissions: accountPermissions ?? state.accountPermissions,
          usage: usage ?? state.usage,
          isAuthenticated: true,
          isHydrated: true,
        })),

      hydrateSession: ({
        user,
        workspaces = [],
        capabilities = null,
        accountPermissions,
        usage = null,
      }) =>
        set((state) => ({
          user,
          workspaces: workspaces.length > 0 ? workspaces : state.workspaces,
          capabilities: capabilities ?? state.capabilities,
          accountPermissions: accountPermissions ?? state.accountPermissions,
          usage: usage ?? state.usage,
          isAuthenticated: true,
          isHydrated: true,
        })),

      logout: () =>
        set({
          token: null,
          user: null,
          workspaces: [],
          capabilities: null,
          accountPermissions: [],
          usage: null,
          isAuthenticated: false,
          isHydrated: true,
        }),
    }),
    {
      name: 'auth-storage',
      version: 2,
      storage: createJSONStorage(() => localStorage),
      migrate: (persisted: unknown) => {
        const root = persisted as {
          state?: Record<string, unknown> & {
            mines?: { id: string; name: string }[];
            businesses?: { id: string; name: string }[];
            workspaces?: { id: string; name: string }[];
            usage?: Record<string, unknown>;
          };
        };
        if (!root?.state) return persisted;
        const s = root.state;
        const mines = s.mines as { id: string; name: string }[] | undefined;
        const businesses = s.businesses as { id: string; name: string }[] | undefined;
        const hasOldLists =
          (mines?.length || businesses?.length) && !(s.workspaces as unknown[])?.length;
        if (hasOldLists) {
          const map = new Map<string, { id: string; name: string }>();
          for (const m of mines || []) map.set(m.id, m);
          for (const b of businesses || []) map.set(b.id, b);
          s.workspaces = [...map.values()];
        }
        delete s.mines;
        delete s.businesses;

        const u = s.usage as Record<string, unknown> | undefined;
        if (u && !u.members) {
          const legacyMembers = (u.membersPerMine ?? u.lavadas) as
            | { used?: number; max?: number }
            | undefined;
          s.usage = {
            workspaces: (u.mines ?? u.workspaces ?? { used: 0, max: 0 }) as {
              used: number;
              max: number;
            },
            members: {
              used: legacyMembers?.used ?? 0,
              max: legacyMembers?.max ?? 0,
            },
          };
        }

        return root;
      },
      partialize: (state) => ({
        token: state.token,
        user: state.user,
        workspaces: state.workspaces,
        capabilities: state.capabilities,
        accountPermissions: state.accountPermissions,
        usage: state.usage,
        isAuthenticated: state.isAuthenticated,
      }),
      onRehydrateStorage: () => (state) => {
        if (state?.token && state?.user) {
          state.isAuthenticated = true;
        }
        if (state) state.isHydrated = true;
      },
    },
  ),
);
